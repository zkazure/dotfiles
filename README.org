** inputrc

#+begin_src conf :tangle ~/.inputrc
  # 开启补全时的文件类型颜色提示 (类似 ls --color)
  set colored-stats on
  set colored-completion-prefix on

  # 在补全列表中显示文件类型符号 (如目录加 /，可执行文件加 *)
  set visible-stats on

  # 补全时忽略大小写 (极其提升输入效率)
  set completion-ignore-case on

  # 遇到多个候选项时，按一次 Tab 直接显示列表，而不是按两次
  set show-all-if-ambiguous on

  # 优化补全前缀的显示逻辑
  set menu-complete-display-prefix on
#+end_src

** mouse

#+begin_src conf :tangle /etc/X11/xorg.conf.d/50-mouse-acceleration.conf
  Section "InputClass"
      Identifier "My Mouse"
      MatchIsPointer "yes"
      # 强制所有鼠标设备使用 Flat 线性模式，关闭加速
      Option "AccelProfile" "flat"
      # 基础速度乘数，0 为 libinput 默认，有效值域通常为 -1 到 1
      Option "AccelSpeed" "0"
  EndSection
#+end_src
